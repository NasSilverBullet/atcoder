def fizzbuzz(n):
    if n % 15 == 0:
        return 'FizzBuzz'
    if n % 5 == 0:
        return 'Buzz'
    if n % 3 == 0:
        return 'Fizz'
    return str(n)


LIMIT = 100
print([fizzbuzz(n) for n in range(1, LIMIT + 1)])
